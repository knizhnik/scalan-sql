template<class O, class I, class K, void(*)(K &key, O const &outer) outerKey, void(*)(K &key, I const &inner) innerKey, class ORdd = RDD<O>, class IRdd = RDD<I>>
class HashJoinRDD< O, I, K, outerKey, innerKey, ORdd, IRdd >

Join two RDDs using hash table. Depending on estimated result size and broadcastJoinThreshold, this RDD either broadcast inner table, either shuffle inner table
